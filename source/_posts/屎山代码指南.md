/*
函数名getNonTopNews应该以动词开头，更好的命名方式可以是fetchNonTopNews。

当使用await关键字时，建议使用try-catch块来处理可能出现的异常。
.then链式调用的方式可读性较差，可以使用async/await语法替代。

可以在fetch请求中添加错误处理，以便在请求失败时返回错误信息。
改进后的代码示例：
*/
async function fetchNonTopNews(root: string, NonTopNews: string): Promise<any> {
    try {
        const url = root + NonTopNews;
        const response = await fetch(url);
        if (!response.ok) {
            throw new Error('Network request failed');
        }
        const data = await response.json();
        return data.result;
    } catch (error) {
        console.error('Fetch non top news error:', error);
        throw error;
    }
}
